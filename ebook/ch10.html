chapter10

Posts & Feeds

text vs link posts, pagination with receipts, and the hot-ranking formula

In this chapter you'll learn ✍️

  • modeling two flavors of post (text / link) in one table
  • JOINs that turn ids into names the frontend actually wants
  • pagination done honestly: LIMIT/OFFSET + a meta object
  • Reddit-style hot sorting — score vs. age, with real math

Two kinds of post, one table

Reddit has text posts (“AMA!”) and link posts (a URL pointing elsewhere). Rather than two tables, we give posts both a body and a url column and add a rule: a post fills one or the other, never both. The validation lives in the handler; the JSON stays clean because both fields are omitempty.

internal/store/posts.go 
package store

import (
	"database/sql"
	"errors"
)

// Post is one submission in a subreddit. Body and URL are omitempty so a
// link post doesn't show "body": "" and a text post doesn't show "url": "".
type Post struct {
	ID           int64  `json:"id"`
	Title        string `json:"title"`
	Body         string `json:"body,omitempty"`
	URL          string `json:"url,omitempty"`
	Score        int64  `json:"score"`
	CommentCount int64  `json:"comment_count"`
	Author       string `json:"author"`
	Subreddit    string `json:"subreddit"`
	CreatedAt    string `json:"created_at"`
}

// PostFilters carries everything a feed query needs: which community
// (0 = all of them), how to sort, and which page to return.
type PostFilters struct {
	SubredditID int64
	Sort        string // "new", "top" or "hot"
	Page        int
	PageSize    int
}

// Meta describes the page of results that came back, so clients can
// render pagination controls.
type Meta struct {
	Page       int   `json:"page"`
	PageSize   int   `json:"page_size"`
	Total      int64 `json:"total"`
	TotalPages int64 `json:"total_pages"`
}

Creating and fetching — JOINs earn their keep

internal/store/posts.go (continued) 
// CreatePost inserts a post and returns it fully hydrated (with author
// and subreddit names attached).
func (s *Store) CreatePost(subredditID, authorID int64, title, body, url string) (*Post, error) {
	var id int64
	err := s.db.QueryRow(`
        INSERT INTO posts (subreddit_id, author_id, title, body, url)
        VALUES (?, ?, ?, ?, ?)
        RETURNING id`,
		subredditID, authorID, title, body, url,
	).Scan(&id)
	if err != nil {
		return nil, err
	}
	return s.PostByID(id)
}
internal/store/posts.go (continued) 
// PostByID fetches a single post. Returns ErrNotFound if it doesn't exist.
func (s *Store) PostByID(id int64) (*Post, error) {
	var p Post
	err := s.db.QueryRow(`
        SELECT p.id, p.title, p.body, p.url, p.score, p.created_at,
               u.username, sr.name,
               (SELECT COUNT(*) FROM comments c WHERE c.post_id = p.id)
        FROM posts p
        JOIN users      u  ON u.id  = p.author_id
        JOIN subreddits sr ON sr.id = p.subreddit_id
        WHERE p.id = ?`,
		id,
	).Scan(&p.ID, &p.Title, &p.Body, &p.URL, &p.Score, &p.CreatedAt,
		&p.Author, &p.Subreddit, &p.CommentCount)

	if errors.Is(err, sql.ErrNoRows) {
		return nil, ErrNotFound
	}
	if err != nil {
		return nil, err
	}
	return &p, nil
}

The posts table stores author_id = 2, but no frontend wants to make a second request to learn that 2 means “linus”. The two JOINs stitch usernames and subreddit names into the row on the way out, and the subquery counts comments — one query, one round-trip, everything a post card on screen needs.

The feed — pagination with receipts

Feeds can't return every post since the dawn of time. Clients ask for a slice — ?page=3&page_size=20 — and SQL slices with LIMIT page_size OFFSET (page-1)*page_size. But a bare array leaves the frontend guessing: is there a next page? how many total? So gopherit always ships a meta receipt alongside:

internal/store/posts.go (continued) 
// ListPosts returns one page of the feed plus pagination metadata.
//
// The "hot" sort is the fun part: score divided by (age in hours + 2)
// raised to 1.5 — fresh posts with a few votes beat old posts with many.
// The +2 stops brand-new posts from dividing by (almost) zero.
func (s *Store) ListPosts(f PostFilters) ([]Post, Meta, error) {
	orderBy := map[string]string{
		"new": "p.id DESC",
		"top": "p.score DESC, p.id DESC",
		"hot": "hotness DESC, p.id DESC",
	}[f.Sort]
	if orderBy == "" {
		orderBy = "p.id DESC"
	}

	// COUNT(*) OVER() is a window function: it stamps the total number of
	// matching rows onto every row, so one query gives us data + total.
	query := `
        SELECT p.id, p.title, p.body, p.url, p.score, p.created_at,
               u.username, sr.name,
               (SELECT COUNT(*) FROM comments c WHERE c.post_id = p.id),
               CAST(p.score AS REAL) /
                   pow((julianday('now') - julianday(p.created_at)) * 24.0 + 2.0, 1.5)
                   AS hotness,
               COUNT(*) OVER() AS total
        FROM posts p
        JOIN users      u  ON u.id  = p.author_id
        JOIN subreddits sr ON sr.id = p.subreddit_id
        WHERE (? = 0 OR p.subreddit_id = ?)
        ORDER BY ` + orderBy + `
        LIMIT ? OFFSET ?`

	offset := (f.Page - 1) * f.PageSize
	rows, err := s.db.Query(query, f.SubredditID, f.SubredditID, f.PageSize, offset)
	if err != nil {
		return nil, Meta{}, err
	}
	defer rows.Close()

	posts := []Post{}
	var total int64
	for rows.Next() {
		var p Post
		var hotness float64 // needed for ORDER BY; not part of the response
		err := rows.Scan(&p.ID, &p.Title, &p.Body, &p.URL, &p.Score, &p.CreatedAt,
			&p.Author, &p.Subreddit, &p.CommentCount, &hotness, &total)
		if err != nil {
			return nil, Meta{}, err
		}
		posts = append(posts, p)
	}
	if err := rows.Err(); err != nil {
		return nil, Meta{}, err
	}

	meta := Meta{
		Page:       f.Page,
		PageSize:   f.PageSize,
		Total:      total,
		TotalPages: (total + int64(f.PageSize) - 1) / int64(f.PageSize),
	}
	return posts, meta, nil
}

Three details deserve a slow read:

  • COUNT(*) OVER() — a window function: it stamps the total number of matching rows onto every returned row, so we get data + total in one query instead of a separate SELECT COUNT(*).
  • WHERE (? = 0 OR p.subreddit_id = ?) — one query serves both the site-wide front page (filter 0 = off) and a single community's feed.
  • The ORDER BY comes from a map — never, ever concatenate user input into SQL. The user's ?sort= string only chooses among our three hard-coded ORDER BY strings; every actual value in the query travels as a ? placeholder. That discipline is what makes SQL injection impossible here.
The “hot” formula, on one napkin 🔥

hotness = score ÷ (hours_old + 2)1.5

“New” shows fresh-but-unproven posts; “top” shows proven-but-possibly-ancient ones. “Hot” wants both: score in the numerator rewards upvotes, age in the denominator drags every post down as it gets old — and the exponent 1.5 makes gravity accelerate: a day-old post needs ~180× the score of a 1-hour-old post to tie. The +2 stops a minutes-old post from dividing by nearly zero and catapulting to #1 with a single vote.

In SQL, julianday('now') - julianday(created_at) gives the age in days (ride those RFC-3339 strings again!), × 24 for hours, then pow(…, 1.5). Real Reddit's algorithm is logarithmic and more elaborate — but this is the honest little sibling of the same idea, and it feels right in practice.

Handlers: create, get, list — twice

Query-string parsing gets its own helper in helpers.go (defaults: hot, page 1, 20 per page, max 100):

cmd/api/helpers.go — add 
// readPostFilters parses ?sort=&page=&page_size= with sane defaults.
func (app *application) readPostFilters(r *http.Request) (store.PostFilters, error) {
	q := r.URL.Query()
	f := store.PostFilters{Sort: "hot", Page: 1, PageSize: 20}

	if s := q.Get("sort"); s != "" {
		if s != "new" && s != "top" && s != "hot" {
			return f, errors.New("sort must be one of: new, top, hot")
		}
		f.Sort = s
	}
	if pg := q.Get("page"); pg != "" {
		n, err := strconv.Atoi(pg)
		if err != nil || n < 1 {
			return f, errors.New("page must be a positive whole number")
		}
		f.Page = n
	}
	if ps := q.Get("page_size"); ps != "" {
		n, err := strconv.Atoi(ps)
		if err != nil || n < 1 || n > 100 {
			return f, errors.New("page_size must be between 1 and 100")
		}
		f.PageSize = n
	}
	return f, nil
}
cmd/api/posts.go 
package main

import (
	"errors"
	"net/http"
	"strings"

	"gopherit/internal/store"
)

// createPostHandler — POST /api/v1/posts  (auth required)
// Body: {"subreddit": "golang", "title": "...", "body": "..."} for a text
// post, or {"subreddit": "...", "title": "...", "url": "https://..."} for
// a link post.
func (app *application) createPostHandler(w http.ResponseWriter, r *http.Request) {
	var input struct {
		Subreddit string `json:"subreddit"`
		Title     string `json:"title"`
		Body      string `json:"body"`
		URL       string `json:"url"`
	}
	if err := app.readJSON(w, r, &input); err != nil {
		app.badRequest(w, err)
		return
	}

	input.Title = strings.TrimSpace(input.Title)
	input.URL = strings.TrimSpace(input.URL)

	p := make(problems)
	p.check(input.Subreddit != "", "subreddit", "must be provided")
	p.check(input.Title != "", "title", "must be provided")
	p.check(len(input.Title) <= 300, "title", "must be at most 300 characters")
	p.check(len(input.Body) <= 40_000, "body", "must be at most 40,000 characters")
	p.check(input.Body == "" || input.URL == "", "url",
		"a post can have a body or a url, not both")
	if input.URL != "" {
		p.check(strings.HasPrefix(input.URL, "http://") || strings.HasPrefix(input.URL, "https://"),
			"url", "must start with http:// or https://")
	}
	if len(p) > 0 {
		app.failedValidation(w, p)
		return
	}

	subreddit, err := app.store.SubredditByName(input.Subreddit)
	if err != nil {
		switch {
		case errors.Is(err, store.ErrNotFound):
			app.failedValidation(w, problems{"subreddit": "no such subreddit"})
		default:
			app.serverError(w, r, err)
		}
		return
	}

	user := app.currentUser(r)
	post, err := app.store.CreatePost(subreddit.ID, user.ID, input.Title, input.Body, input.URL)
	if err != nil {
		app.serverError(w, r, err)
		return
	}

	app.writeJSON(w, http.StatusCreated, envelope{"post": post})
}

// getPostHandler — GET /api/v1/posts/{id}
// Returns the post AND its full comment tree in one response.
func (app *application) getPostHandler(w http.ResponseWriter, r *http.Request) {
	id, err := app.readIDParam(r)
	if err != nil {
		app.notFound(w)
		return
	}

	post, err := app.store.PostByID(id)
	if err != nil {
		switch {
		case errors.Is(err, store.ErrNotFound):
			app.notFound(w)
		default:
			app.serverError(w, r, err)
		}
		return
	}

	comments, err := app.store.CommentsForPost(id)
	if err != nil {
		app.serverError(w, r, err)
		return
	}

	app.writeJSON(w, http.StatusOK, envelope{"post": post, "comments": comments})
}

// listPostsHandler — GET /api/v1/posts?sort=hot&page=1&page_size=20
// The site-wide front page.
func (app *application) listPostsHandler(w http.ResponseWriter, r *http.Request) {
	filters, err := app.readPostFilters(r)
	if err != nil {
		app.badRequest(w, err)
		return
	}

	posts, meta, err := app.store.ListPosts(filters)
	if err != nil {
		app.serverError(w, r, err)
		return
	}

	app.writeJSON(w, http.StatusOK, envelope{"posts": posts, "meta": meta})
}

// listSubredditPostsHandler — GET /api/v1/subreddits/{name}/posts
// Same as the front page, but scoped to one community.
func (app *application) listSubredditPostsHandler(w http.ResponseWriter, r *http.Request) {
	subreddit, err := app.store.SubredditByName(r.PathValue("name"))
	if err != nil {
		switch {
		case errors.Is(err, store.ErrNotFound):
			app.notFound(w)
		default:
			app.serverError(w, r, err)
		}
		return
	}

	filters, err := app.readPostFilters(r)
	if err != nil {
		app.badRequest(w, err)
		return
	}
	filters.SubredditID = subreddit.ID

	posts, meta, err := app.store.ListPosts(filters)
	if err != nil {
		app.serverError(w, r, err)
		return
	}

	app.writeJSON(w, http.StatusOK, envelope{"posts": posts, "meta": meta})
}

Note the validation twist in createPostHandler: “no such subreddit” is reported as a 422 on the subreddit field, exactly like a too-long title — from the client's perspective, both are “fix your input”, not “resource missing”.

Pin to your brain 📌

  • Variant data (text/link) can share a table; enforce the either/or in validation.
  • JOIN names into responses — don't make clients do id lookups.
  • Pagination = LIMIT/OFFSET + a meta receipt (COUNT(*) OVER() gets both at once).
  • User input may choose between hard-coded SQL strings; it must never become SQL.
  • hot = score ÷ (hours+2)1.5 — upvotes lift, gravity pulls, +2 protects newborns.
